EVENT > Validare un Form su Eventi diversi da Submit
















<script>
validation.form('btnValida',{event:'blur',label:'top',callBack:function(a,f){if(a.length < 1)f.submit()}});
</script>

< form action="./index2.html" class="validateForm6">
<input type="text" value="numerico" valid="{type:'number',message:'campo errato numerico'}"/>
<input type="text" value="numerico" valid="{type:'email',message:'campo errato ecc.ecc'}"/>
<input type="text" value="numerico" valid="{type:'alfa',message:'campo errato ecc.ecc'}"/>
<input type="text" value="numerico" valid="{type:'password',message:'campo errato password'}"/>
<input type="button" value="Valida e abilita submit sul blur" class="btnValida" />
< /form>
EVENT > abbinare ulteriori eventi ad elementi
















<script>
validation.form('validateForm7',{callBack:function(a){alert(a[0].message)}});
validation.captured({nodeClass:"myEvent",attach:[{fn:template.base.evidenzia,event:"focus blur"}]});
</script>

< form action="./index2.html" class="validateForm7">
<input type="text" value="" class="myEvent" valid="{type:'number',message:'campo numerico errato'}"/>
<input type="text" value="" class="myEvent" valid="{type:'email',message:'campo email errato'}"/>
<input type="text" value="" class="myEvent" valid="{type:'alfa',message:'campo testo errato'}"/>
<input type="text" value="" class="myEvent" valid="{type:'password',message:'campo password errato'}"/>
<input type="submit" value="send" />
< /form>
